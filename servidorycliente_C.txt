#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>

#define PORT 8080

int main() {
    int server_fd, client_fd;
    struct sockaddr_in address;
    int addrlen = sizeof(address);
    char buffer[1024] = {0};
    
    // Creo socket
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    
    // Configuro direcci√≥n
    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);
    
    // Vincular y escuchar
    bind(server_fd, (struct sockaddr*)&address, sizeof(address));
    listen(server_fd, 3);
    
    printf("Servidor esperando conexion...\n");
    
    // Aceptar cliente
    client_fd = accept(server_fd, (struct sockaddr*)&address, (socklen_t*)&addrlen);
    printf("Cliente conectado!\n");
    
    // Leer mensaje del cliente
    read(client_fd, buffer, 1024);
    printf("Cliente: %s\n", buffer);
    
    // Enviar respuesta
    char* respuesta = "Hola desde el servidor";
    send(client_fd, respuesta, strlen(respuesta), 0);
    printf("Respuesta enviada\n");
    
    // Cerrar
    close(client_fd);
    close(server_fd);
    
    return 0;
}

#include <stdio.h>
#include <string.h>
#include <sys/socket.h>
#include <arpa/inet.h>
#include <unistd.h>

#define PORT 8080

int main() {
    int client_fd;
    struct sockaddr_in serv_addr;
    char buffer[1024] = {0};
    
    // Creo socket
    client_fd = socket(AF_INET, SOCK_STREAM, 0);
    
    // Configuro servidor
    serv_addr.sin_family = AF_INET;
    serv_addr.sin_port = htons(PORT);
    inet_pton(AF_INET, "127.0.0.1", &serv_addr.sin_addr);
    
    // Conecto
    connect(client_fd, (struct sockaddr*)&serv_addr, sizeof(serv_addr));
    printf("Conectado al servidor\n");
    
    // Enviar mensaje
    char* mensaje = "Hola desde el cliente";
    send(client_fd, mensaje, strlen(mensaje), 0);
    printf("Mensaje enviado\n");
    
    // Recibir respuesta
    read(client_fd, buffer, 1024);
    printf("Servidor dice: %s\n", buffer);
    
    // Cerrar
    close(client_fd);
    
    return 0;
}
